```ini
[VideoDecoding]

[Splitter]
EnableMKV=True
EnableFLV=True
EnableOGM=True
EnableAVI=True
EnableMP4=True
SubtitleStreamHandler=DirectVobSub
OSDEnabled=false

[AdvancedFilters]
RainbowFix=true
DenoiseLevel=Low
SharpeningStrength=0.3
```
